package com.example.smokeduel;

public class ScoringPipeline {
    public static String scoreDuel(int player1Score, int player2Score) {
        if (player1Score > player2Score) {
            return "Player 1 wins";
        } else if (player1Score < player2Score) {
            return "Player 2 wins";
        } else {
            return "It's a tie";
        }
    }

    public static void main(String[] args) {
        System.out.println(scoreDuel(10, 20));
        System.out.println(scoreDuel(30, 40));
        System.out.println(scoreDuel(50, 50));
    }
}
